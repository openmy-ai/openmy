```python
def test_run_silero_vad_graceful_on_missing_dep(monkeypatch):
    """silero-vad 未安装时返回空列表不报错"""
    import builtins
    original_import = builtins.__import__
    def mock_import(name, *args, **kwargs):
        if name in ('silero_vad', 'torch'):
            raise ImportError(f"mock: {name} not installed")
        return original_import(name, *args, **kwargs)

    monkeypatch.setattr(builtins, '__import__', mock_import)
    from openmy.services.ingest.audio_pipeline import run_silero_vad
    # 重新加载模块以触发 ImportError
    result = run_silero_vad(Path("/nonexistent/audio.wav"))
    assert result == []
```
