```javascript
// 缓存已解码的波形数据
const waveformCache = new Map();

/**
 * 从音频 URL 获取波形数据
 * @param {string} audioUrl — 音频文件 URL
 * @returns {Promise<Float32Array>} — 归一化的波形采样数据
 */
export async function fetchWaveformData(audioUrl) {
  if (waveformCache.has(audioUrl)) return waveformCache.get(audioUrl);
  const response = await fetch(audioUrl);
  const arrayBuffer = await response.arrayBuffer();
  const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
  const audioBuffer = await audioCtx.decodeAudioData(arrayBuffer);
  const channelData = audioBuffer.getChannelData(0);

  // 降采样到每秒 10 个点（足够画波形）
  const samplesPerPoint = Math.floor(audioBuffer.sampleRate / 10);
  const points = Math.ceil(channelData.length / samplesPerPoint);
  const waveform = new Float32Array(points);
  for (let i = 0; i < points; i++) {
    let sum = 0;
    const start = i * samplesPerPoint;
    const end = Math.min(start + samplesPerPoint, channelData.length);
    for (let j = start; j < end; j++) {
      sum += Math.abs(channelData[j]);
    }
    waveform[i] = sum / (end - start);
  }

  waveformCache.set(audioUrl, { waveform, duration: audioBuffer.duration });
  await audioCtx.close();
  return { waveform, duration: audioBuffer.duration };
}

/**
 * 在 Canvas 上绘制波形
 * @param {HTMLCanvasElement} canvas
 * @param {Float32Array} waveform — 波形数据
 * @param {number} duration — 音频总时长（秒）
 * @param {Array} speechSegments — [{start, end}] VAD 语音段
 * @param {number} currentTime — 当前播放进度（秒）
 */
export function renderWaveform(canvas, waveform, duration, speechSegments = [], currentTime = 0) {
  const ctx = canvas.getContext('2d');
  const { width, height } = canvas;
  const dpr = window.devicePixelRatio || 1;
  canvas.width = width * dpr;
  canvas.height = height * dpr;
  ctx.scale(dpr, dpr);
  ctx.clearRect(0, 0, width, height);

  const barWidth = width / waveform.length;
  const mid = height / 2;
  const maxAmp = Math.max(...waveform) || 1;

  // 预计算 VAD 区间（按波形索引）
  const pointsPerSecond = waveform.length / duration;
  const vadRanges = speechSegments.map(s => ({
    startIdx: Math.floor(s.start * pointsPerSecond),
    endIdx: Math.ceil(s.end * pointsPerSecond),
  }));
  const progressIdx = Math.floor((currentTime / duration) * waveform.length);

  for (let i = 0; i < waveform.length; i++) {
    const amp = (waveform[i] / maxAmp) * mid * 0.9;
    const inSpeech = vadRanges.some(r => i >= r.startIdx && i <= r.endIdx);
    const isPlayed = i <= progressIdx;

    if (isPlayed) {
      ctx.fillStyle = inSpeech ? 'var(--waveform-played-speech, #4f9cf7)' : 'var(--waveform-played-silence, #3a6ea5)';
    } else {
      ctx.fillStyle = inSpeech ? 'var(--waveform-speech, rgba(79,156,247,0.5))' : 'var(--waveform-silence, rgba(255,255,255,0.15))';
    }

    ctx.fillRect(i * barWidth, mid - amp, Math.max(barWidth - 0.5, 1), amp * 2);
  }

  // 播放进度线
  const progressX = (currentTime / duration) * width;
  ctx.strokeStyle = 'var(--waveform-cursor, #fff)';
  ctx.lineWidth = 2;
  ctx.beginPath();
  ctx.moveTo(progressX, 0);
  ctx.lineTo(progressX, height);
  ctx.stroke();
}

/**
 * 给 Canvas 绑定拖动事件
 * @param {HTMLCanvasElement} canvas
 * @param {number} duration — 音频总时长
 * @param {function} onSeek — 回调，参数为目标秒数
 */
export function bindWaveformSeek(canvas, duration, onSeek) {
  let dragging = false;
  const getTime = (e) => {
    const rect = canvas.getBoundingClientRect();
    const x = (e.clientX || e.touches?.[0]?.clientX || 0) - rect.left;
    return Math.max(0, Math.min(duration, (x / rect.width) * duration));
  };
  canvas.addEventListener('mousedown', (e) => { dragging = true; onSeek(getTime(e)); });
  canvas.addEventListener('mousemove', (e) => { if (dragging) onSeek(getTime(e)); });
  canvas.addEventListener('mouseup', () => { dragging = false; });
  canvas.addEventListener('mouseleave', () => { dragging = false; });
  canvas.addEventListener('touchstart', (e) => { dragging = true; onSeek(getTime(e)); }, { passive: true });
  canvas.addEventListener('touchmove', (e) => { if (dragging) onSeek(getTime(e)); }, { passive: true });
  canvas.addEventListener('touchend', () => { dragging = false; });
}
```
