```python
def run_silero_vad(audio_path: Path) -> list[dict[str, float]]:
    """用 Silero VAD 检测音频中的人声段，返回 [{start, end}] 秒数列表。"""
    try:
        import torch
        from silero_vad import load_silero_vad, get_speech_timestamps
    except ImportError:
        return []  # silero-vad 未安装，降级为空

    # 先转成 16kHz mono wav
    tmp_wav = Path(tempfile.mktemp(suffix=".wav"))
    try:
        run_ffmpeg([
            "-i", str(audio_path),
            "-ar", "16000", "-ac", "1",
            "-f", "wav", "-acodec", "pcm_s16le",
            str(tmp_wav), "-y", "-loglevel", "error",
        ])

        # 手动加载 wav（绕过 torchaudio.sox 兼容问题）
        import wave
        import struct
        with wave.open(str(tmp_wav), 'rb') as wf:
            n_frames = wf.getnframes()
            raw_data = wf.readframes(n_frames)
            sr = wf.getframerate()
        samples = struct.unpack(f'<{n_frames}h', raw_data)
        wav_tensor = torch.FloatTensor(samples) / 32768.0

        model = load_silero_vad()
        speech_timestamps = get_speech_timestamps(
            wav_tensor, model, return_seconds=True, sampling_rate=sr,
        )
        return [{"start": round(s["start"], 2), "end": round(s["end"], 2)} for s in speech_timestamps]
    except Exception:
        return []
    finally:
        if tmp_wav.exists():
            tmp_wav.unlink()
```

关键点：
- lazy import torch 和 silero_vad，未安装时返回空列表
- 用 wave + struct 绕过 torchaudio.sox 兼容问题
- 异常保护，任何失败返回空列表不影响 pipeline
